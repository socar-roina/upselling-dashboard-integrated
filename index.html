한계 (논리상 미리 박아둠)

• 인과: 정규화된 PF는 대조군이 없어 넛지가 '추가로' 유발한 비율은 정밀 측정 불가(원래 살 사람 포함)
노출당과 수락당 비교엔 PF와 시간연장에 동일 적용이라 상대 비교는 유효
시간연장은 ABT라 대조군 기반 정밀 인과는 별도 가능
• 단위: 흐름과 비교표 모두 예약 건 기준으로 통일
• 미노출 등급: pf_type은 결제 결과 등급
'이미 PF0'는 결과 PF0(자기부담금 0원)면서 미노출인 건으로 진입 시점 등급의 근사
• 최근 지연: 6/13~6/14 수락은 결제 D+1 반영으로 과소집계 가능

• 시간연장과 차종 수락/매출: socar-data.dbt.stg_tianjin_replica__paid_info (paid_type='upselling_discount', 차종은 6/10 이후분)

쿼리 (미노출 분해)

-- 예약을 노출/미노출 × 보험등급으로 분해
WITH v AS (
  SELECT DISTINCT DATE(event_at_kst) d, member_id
  FROM `socar-data.app_web_log.socar_app_web_log`
  WHERE page_name='upsell_bottomsheet' AND event_name='view'
    AND event_at_kst >= DATETIME '2026-05-21 14:00:00'
    AND event_at_kst <  DATETIME '2026-06-15 00:00:00'
),
r AS (
  SELECT DATE(created_at_kst) d, member_id, pf_type, utime, region1
  FROM `socar-data.socar_biz_profit.profit_socar_reservation`
  WHERE created_at_kst >= DATETIME '2026-05-21 14:00:00'
    AND created_at_kst <  DATETIME '2026-06-15 00:00:00'
)
SELECT IF(v.member_id IS NULL,'미노출','노출') expose, r.pf_type,
  COUNT(*) rsv, COUNT(DISTINCT r.member_id) uu
FROM r LEFT JOIN v ON r.d=v.d AND r.member_id=v.member_id
GROUP BY 1,2;

쿼리 (아이템 분류, 양 포맷)

-- 6/10 09:42 포맷 변경(upselling_type -> ut) 양쪽 파싱
SELECT
  CASE
    WHEN REGEXP_CONTAINS(form_data, r'upselling_type[\\:"]+pfNudge')
         OR REGEXP_CONTAINS(form_data, r'ut[\\:"]+p[\\}",]')  THEN 'PF'
    WHEN REGEXP_CONTAINS(form_data, r'upselling_type[\\:"]+intervalUpselling')
         OR REGEXP_CONTAINS(form_data, r'ut[\\:"]+i[\\}",]')  THEN '시간연장'
    WHEN REGEXP_CONTAINS(form_data, r'upselling_type[\\:"]+carUpgradeUpselling')
         OR REGEXP_CONTAINS(form_data, r'ut[\\:"]+c[\\}",]')  THEN '차종'
  END AS item,
  COUNT(DISTINCT FORMAT('%t-%d', DATE(event_at_kst), member_id)) AS view_member_day
FROM `socar-data.app_web_log.socar_app_web_log`
WHERE page_name='upsell_bottomsheet' AND event_name='view'
  AND event_at_kst >= DATETIME '2026-05-21 14:00:00'
  AND event_at_kst <  DATETIME '2026-06-15 00:00:00'
GROUP BY 1;
